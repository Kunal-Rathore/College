class Solution {
public:
    int numberOfBeams(vector<string>& bank) {
 
// Approach -> i)traverse each string and count the devices if its greater then 0 then store in ans by multiplyin to previous no. of devices which will be stored in storeCount 



               int ans = 0;
//i)
int countt =0;
int storeCount =0;
        for(auto str: bank)
        {
            
             countt = count(str.begin(), str.end(),'1');

            if(countt>0)
            {
               ans += countt * storeCount;   // if there is only one row or 0 row with devices then there will be 0 laser beams in that case ans will also be zero so edge case also satisfies here
                storeCount = countt;
                countt = 0;
            }
         }
  
return ans;
    }
};
